A<-c(12,15,14,10,13)
B<-c(18,20,17,22,19)
data=c(A,B)
group=factor(c(rep('A',5),rep('B',5)))
median_val=median(data)
tbl=table(group,data > median_val)
fisher.test(tbl)
chisq.test(table(group, data > median_val))
boxplot(A,B,names=c('A','B'),main='boxplot')
hist(A,col='lightblue')
hist(B,col='pink',add=TRUE)
plot(density(A),col='blue')
lines(density(B),col='red')
abline(h=median_val,col='red')
