# Data
data <- c(12,15,18,20,22,24,27)
M0 <- 20
alpha <- 0.05

# Remove ties
signs <- sign(data - M0)
signs <- signs[signs != 0]

# Test statistic
S <- sum(signs > 0)
n <- length(signs)

# p-value (two-tailed)
p_value <- 2 * pbinom(min(S, n - S), n, 0.5)

cat("Number of positive signs:", S, "\n")
cat("Sample size (without ties):", n, "\n")
cat("p-value:", p_value, "\n")

# Decision
if(p_value < alpha){
  cat("Conclusion: Reject H0\n")
  cat("Median is significantly different from", M0)
} else {
  cat("Conclusion: Fail to Reject H0\n")
  cat("Median is not significantly different from", M0)
}

# Visualization (optional)
x <- 0:n
prob <- dbinom(x, n, 0.5)

barplot(prob, names.arg = x,
        main = "Binomial Distribution (Sign Test)",
        xlab = "Number of Positive Signs",
        ylab = "Probability")

# Sign visualization
plot(data, pch=19,
     main="Sign Test Visualization",
     ylab="Values")

abline(h = M0, lty = 2)
