#infix to postfix
class Stack:
    def __init__(self, capacity):
        self.stack = []
    
    def push(self, item):
        self.stack.append(item)

    def pop(self):
        if self.stack:
            return self.stack.pop()

    def peek(self):
        if self.stack:
            return self.stack[-1]

    def is_empty(self):
        return len(self.stack) == 0


def precedence(op):
    if op in '+-':
        return 1
    if op in '*/':
        return 2
    return 0


def infix_to_postfix(infix):
    stack = Stack(len(infix))
    postfix = []

    for char in infix:
        if char.isalnum():
            postfix.append(char)

        elif char == '(':
            stack.push(char)

        elif char == ')':
            while not stack.is_empty() and stack.peek() != '(':
                postfix.append(stack.pop())
            stack.pop()

        else:
            while (not stack.is_empty() and stack.peek() != '(' and
                   precedence(char) <= precedence(stack.peek())):
                postfix.append(stack.pop())
            stack.push(char)

    while not stack.is_empty():
        postfix.append(stack.pop())

    return ''.join(postfix)
expressions = ["A+B*C", "(A+B)*C", "A+B*C-D", "A*B+C*D"]

for expr in expressions:
    print(expr, "->", infix_to_postfix(expr))
