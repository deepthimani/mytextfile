# Create contingency table
data1 <- matrix(c(40,60,20,80), nrow=2, byrow=TRUE)
rownames(data1) <- c("Smokers", "NonSmokers")
colnames(data1) <- c("LungDisease", "NoLungDisease")
data1

# Chi-square test
test1 <- chisq.test(data1, correct=FALSE)
test1

# Conclusion using Critical Value Method
df1 <- test1$parameter
critical1 <- qchisq(0.95, df1)

cat("\nCritical Value =", critical1, "\n")

if(test1$statistic > critical1){
  cat("Decision (Critical Value Method): Reject H0\n")
} else {
  cat("Decision (Critical Value Method): Fail to Reject H0\n")
}

# Side-by-side bar plot
barplot(data1,
        beside=TRUE,
        col=c("steelblue","tomato"),
        legend=rownames(data1),
        main="Smoking vs Lung Disease",
        ylab="Frequency")

# Mosaic plot
mosaicplot(data1,
           color=TRUE,
           main="Mosaic Plot: Smoking & Lung Disease")

# Association plot
assocplot(data1,
          main="Association Plot: Smoking & Lung Disease")
